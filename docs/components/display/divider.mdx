# Component | Divider

Visually separate content with a horizontal or vertical line. Dividers are useful for creating logical groupings and improving the scannability of your UI.

<IFrame type="components-display-divider" heightPx={500} />

## API reference

### Function signature

<Tabs groupId="sdkLanguage">
    <TabItem value="node" label="TypeScript / JavaScript">
    ```ts
    ui.divider(
        options?: Partial<{
            orientation: "horizontal" | "vertical",
            thickness: "thin" | "medium" | "thick",
            style: Style
        }>
    )
    ```
    </TabItem>
    <TabItem value="python" label="Python">
    ```python
    ui.divider(
        *,
        orientation: Literal["horizontal", "vertical"] = "horizontal",
        thickness: Literal["thin", "medium", "thick"] = "medium",
        style: Style | None = None
    )
    ```
    </TabItem>
</Tabs>

### Parameters

<Parameter
    name="orientation"
    required={false}
    type="literal"
    description={lang =>
        <>
            <p>
                The orientation of the divider. Defaults to <code>"horizontal"</code>.
            </p>
            <p>Options:</p>
            <ul>
                <li>
                    <code>"horizontal"</code>: A horizontal line.
                </li>
                <li>
                    <code>"vertical"</code>: A vertical line. Useful when using the divider is inside a row component.
                </li>
            </ul>
        </>
    }
/>

<Parameter
    name="thickness"
    required={false}
    type="literal"
    description={lang =>
        <>
            <p>
                The thickness of the divider line. Defaults to <code>"thin"</code>.
            </p>
            <p>Options:</p>
            <ul>
                <li>
                    <code>"thin"</code>: <code>1px</code>
                </li>
                <li>
                    <code>"medium"</code>: <code>2px</code>
                </li>
                <li>
                    <code>"thick"</code>: <code>4px</code>
                </li>
            </ul>
        </>
    }
/>
