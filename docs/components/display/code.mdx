## API reference

### Function signature

<Tabs groupId="sdkLanguage">
    <TabItem value="node" label="TypeScript / JavaScript">
    ```ts
    ui.code(
        text: string,
        options?: {
            label?: string,
            description?: string,
            lang?: "ada" | "clojure" | "coffeescript" | /* ... */ | "r"
            style?: Style
        }
    )
    ```
    </TabItem>
    <TabItem value="python" label="Python">
    ```python
    ui.code(
        text: str,
        *,
        label: str = None,
        description: str = None,
        lang: Literal["ada", "clojure", "coffeescript", /* ... */, "r"] | None = None,
        style: Style | None = None,
    )
    ```
    </TabItem>
</Tabs>
