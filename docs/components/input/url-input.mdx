# Component | URL Input

Render a url input box.

<Tabs groupId="sdkLanguage">
    <TabItem value="node" label="TypeScript / JavaScript">
    ```js
    ui.urlInput("url-input-key", { 
        label: "Enter a url",
        onEnter: (value) => console.log(value)
    })
    ```
    </TabItem>
    <TabItem value="python" label="Python">
    ```python
    ui.url_input(
        "url-input-key",
        label="Enter a url",
        on_enter=lambda value: print(value)
    )
    ```
    </TabItem>
</Tabs>

<Tabs groupId="sdkLanguage">

    ui.urlInput(
        id: string,
        properties?: Partial<{
            label: string,
            description: string,
            required: boolean,
            initialValue: string | null,
            validate: (value: string | null) => string | void,

        description: str | None = None,
        required: bool = True,
        initial_value: str | None = None,
        validate: Callable[[str | None], str | None] | None = None,
