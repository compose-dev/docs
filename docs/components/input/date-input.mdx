## API reference

### Function signature

<Tabs groupId="sdkLanguage">
    <TabItem value="node" label="TypeScript / JavaScript">
    ```js
    /**
     * One of the following:
     * - A JS Date object in UTC
     * - A human readable object with year, month (1-12), and day (1-31)
     * - `null` to ignore the parameter and use the default value
     */
    type DateInput = 
        | {
            year: number;
            month: number; // 1-12
            day: number; // 1-31
        }
        | Date // Assumes UTC for the given date
        | null

    /**
     * An object with human readable values for year, month (1-12), and day (1-31)
     * and a JS Date object in UTC for the selected date.
     */
    type DateOutput = 
        | {
            year: number;
            month: number; // 1-12
            day: number; // 1-31
            jsDate: Date; // UTC midnight of the selected date
        }
        | null // Passes `null` if no date was selected

    ui.dateInput(
        id: string,
        properties: Partial<{
            label: string,
            description: string,
            required: boolean,
            min: DateInput,
            max: DateInput,
            initialValue: DateInput,
            validate: (value: DateOutput | null) => string | void,
            onEnter: (value: DateOutput | null) => void,
            style: Style | null
        }>
    )
    ```
    </TabItem>
    <TabItem value="python" label="Python">
    ```python
    DateInputType = datetime.date | datetime.datetime | None
    DateOutputType = datetime.date | None

    ui.date_input(
        id: str,
        *,
        label: str = None,
        description: str = None,
        required: bool = True,
        min: DateInputType | None = None,
        max: DateInputType | None = None,
        initial_value: DateInputType | None = None,
        validate: Callable[[DateOutputType | None], str | None] = None,
        on_enter: Callable[[DateOutputType | None], None] = None,
        style: Style | None = None
    )
    ```
    </TabItem>
</Tabs>
