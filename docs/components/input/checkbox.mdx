# Component | Checkbox

Render a checkbox.

<Tabs groupId="sdkLanguage">
    <TabItem value="node" label="TypeScript / JavaScript">
    ```js
    ui.checkbox("checkbox-key", { 
        label: "I agree that Compose is the best thing since sliced bread",
        onChange: (value) => console.log(value)
    })
    ```
    </TabItem>
    <TabItem value="python" label="Python">
    ```python
    ui.checkbox(
        "checkbox-key",
        label="I agree that Compose is the best thing since sliced bread",
        on_change=lambda value: print(value)
    )
    ```
    </TabItem>
</Tabs>

## Function signature

<Tabs groupId="sdkLanguage">
    <TabItem value="node" label="TypeScript / JavaScript">
    ```js
    ui.checkbox(
        id: string,
        properties?: Partial<{
            label: string,
            description: string,
            initialValue: boolean, // Default: false
            validate: (value: boolean) => string | void,
            onChange: (value: boolean) => void,
            style: Style | null
        }>
    )
    ```
    </TabItem>
    <TabItem value="python" label="Python">
    ```python
    ui.checkbox(
        id,
        *,
        label=None,
        description=None,
        initial_value=False,
        validate=None,
        on_change=None,
        style=None
    )
    ```
    </TabItem>
</Tabs>
