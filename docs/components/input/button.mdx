## Appearance

Use the `appearance` property to easily style the button from a predefined set of options.

<Tabs groupId="sdkLanguage">
    <TabItem value="node" label="TypeScript / JavaScript">
    ```js
    ui.button("1", { label: "Primary", appearance: "primary" })
    ui.button("2", { label: "Outline", appearance: "outline" })
    ui.button("3", { label: "Warning", appearance: "warning" })
    ui.button("4", { label: "Danger", appearance: "danger" })
    ```
    </TabItem>
    <TabItem value="python" label="Python">
    ```python
    ui.button("1", label="Primary", appearance="primary")
    ui.button("2", label="Outline", appearance="outline")
    ui.button("3", label="Warning", appearance="warning")
    ui.button("4", label="Danger", appearance="danger")
    ```
    </TabItem>
</Tabs>

<IFrame type="components-input-button-appearance" heightPx={300} />

Or, use the `style` property to apply custom styles to the button.

<Tabs groupId="sdkLanguage">
    <TabItem value="node" label="TypeScript / JavaScript">
    ```js
    ui.button("custom", { 
        label: "Custom button", 
        style: { 
            backgroundColor: "black", 
            color: "white", 
            padding: "10px 20px", 
            borderRadius: "999px" 
        } 
    })
    ```
    </TabItem>
    <TabItem value="python" label="Python">
    ```python
    ui.button(
        "custom", 
        label="Custom button", 
        style={
            "backgroundColor": "black", 
            "color": "white", 
            "padding": "10px 20px", 
            "borderRadius": "999px"
        }
    )
    ```
    </TabItem>
</Tabs>

## API reference

### Function signature

<Tabs groupId="sdkLanguage">
    <TabItem value="node" label="TypeScript / JavaScript">
    ```js
    ui.button(
        id: string,
        properties?: Partial<{
            label: string,
            appearance: "primary" | "outline" | "warning" | "danger",
            onClick: () => void,
            style: Style
        }>
    )
    ```
    </TabItem>
    <TabItem value="python" label="Python">
    ```python
    ui.button(
        id: str,
        *,
        label: str | None = None,
        appearance: Literal["primary", "outline", "warning", "danger"] = "primary",
        on_click: Callable[[], None] | None = None,
        style: Style | None = None
    )
    ```
    </TabItem>
</Tabs>
